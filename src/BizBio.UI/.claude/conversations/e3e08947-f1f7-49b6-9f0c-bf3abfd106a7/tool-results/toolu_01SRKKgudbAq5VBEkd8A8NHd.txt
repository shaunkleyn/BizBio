    22→          p && p.name === 'nuxt:components:imports'
    23→        );
    24→

    26→          const originalPlugin = config.plugins[componentsIndex] as any;
    27→

    30→            const originalTransformInclude = originalPlugin.transformInclude;
    31→            originalPlugin.transformInclude = function(id: string) {
    32→              // Skip virtual modules that cause path issues

    39→                id.startsWith('plugin-') // Plugin virtual modules

    41→                return false;
