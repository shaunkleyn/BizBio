    10→[Route("api/v1/library/extras")]

    12→[Authorize]
    13→public class ExtrasController : ControllerBase
    14→{
    15→    private readonly ApplicationDbContext _context;
    16→    private readonly ILogger<ExtrasController> _logger;
    17→
    18→    public ExtrasController(ApplicationDbContext context, ILogger<ExtrasController> logger)
    19→    {
    20→        _context = context;
    21→        _logger = logger;
    22→    }
    23→
    24→    private int GetUserId() => int.Parse(User.FindFirst(ClaimTypes.NameIdentifier)!.Value);
    25→
    26→    /// <summary>
    27→    /// Get all extras for the current user
    28→    /// </summary>
    29→    [HttpGet]
    30→    public async Task<IActionResult> GetExtras()
    31→    {
    32→        try
    33→        {
    34→            var userId = GetUserId();
    35→            var extras = await _context.CatalogItemExtras
    36→                .Where(e => e.UserId == userId && e.CatalogId == null && e.IsActive)
    37→                .OrderBy(e => e.DisplayOrder)
    38→                .ThenBy(e => e.Name)
    39→                .ToListAsync();
    40→
    41→            return Ok(new { success = true, data = extras });
    42→        }
    43→        catch (Exception ex)
    44→        {
    45→            _logger.LogError(ex, "Error fetching extras");
    46→            return StatusCode(500, new { success = false, error = "Failed to fetch extras" });
    47→        }
    48→    }
    49→
    50→    /// <summary>
    51→    /// Get a single extra by ID
    52→    /// </summary>
    53→    [HttpGet("{id}")]
    54→    public async Task<IActionResult> GetExtra(int id)
    55→    {
    56→        try
    57→        {
    58→            var userId = GetUserId();
    59→            var extra = await _context.CatalogItemExtras
    60→                .FirstOrDefaultAsync(e => e.Id == id && e.UserId == userId && e.IsActive);
    61→
    62→            if (extra == null)
    63→                return NotFound(new { success = false, error = "Extra not found" });
    64→
    65→            return Ok(new { success = true, data = extra });
    66→        }
    67→        catch (Exception ex)
    68→        {
    69→            _logger.LogError(ex, "Error fetching extra {Id}", id);
    70→            return StatusCode(500, new { success = false, error = "Failed to fetch extra" });
    71→        }
    72→    }
    73→
    74→    /// <summary>
    75→    /// Create a new extra
    76→    /// </summary>
    77→    [HttpPost]
    78→    public async Task<IActionResult> CreateExtra([FromBody] CreateExtraDto dto)
    79→    {
    80→        try
    81→        {
    82→            var userId = GetUserId();
    83→            var userIdString = userId.ToString();
    84→
    85→            var extra = new CatalogItemExtra
    86→            {
    87→                UserId = userId,
    88→                Name = dto.Name,
    89→                Description = dto.Description,
    90→                Code = dto.Code,
    91→                BasePrice = dto.BasePrice,
    92→                ImageUrl = dto.ImageUrl,
    93→                DisplayOrder = dto.DisplayOrder,
    94→                IsActive = true,
    95→                CreatedAt = DateTime.UtcNow,
    96→                UpdatedAt = DateTime.UtcNow,
    97→                CreatedBy = userIdString,
    98→                UpdatedBy = userIdString
    99→            };
   100→
   101→            _context.CatalogItemExtras.Add(extra);
   102→            await _context.SaveChangesAsync();
   103→
   104→            return CreatedAtAction(nameof(GetExtra), new { id = extra.Id },
   105→                new { success = true, data = extra });
   106→        }
   107→        catch (Exception ex)
   108→        {
   109→            _logger.LogError(ex, "Error creating extra");
   110→            return StatusCode(500, new { success = false, error = "Failed to create extra" });
   111→        }

   130→            extra.Name = dto.Name;

   132→            extra.Code = dto.Code;

   134→            extra.ImageUrl = dto.ImageUrl;

   143→        catch (Exception ex)

   159→            var userIdString = userId.ToString();
