   247→const api = useApi()
   248→const loading = ref(true)
   249→const error = ref('')
   250→const extraGroups = ref<any[]>([])

   252→const showCreateModal = ref(false)
   253→const editingGroup = ref<any>(null)
   254→const saving = ref(false)

   256→const formData = ref({
   257→  Name: '',
   258→  Description: '',
   259→  MinRequired: 0,

   261→  AllowMultipleQuantities: true,

   265→
   266→const fetchExtraGroups = async () => {
   267→  loading.value = true
   268→  error.value = ''
   269→  try {
   270→    const response = await api.get('/library/extra-groups')
   271→    console.log('Extra groups response:', response);
   272→    if (response.success) {
   273→      // Ensure we always set an array
   274→      if (Array.isArray(response.data)) {
   275→        extraGroups.value = response.data
   276→      } else if (response.data && Array.isArray(response.data.extraGroups)) {
   277→        extraGroups.value = response.data.extraGroups
   278→      } else {
   279→        console.warn('Unexpected response structure:', response.data)
   280→        extraGroups.value = []
   281→      }
   282→    } else {
   283→      error.value = 'Failed to load extra groups'
   284→      extraGroups.value = []
   285→    }
   286→  } catch (err: any) {
   287→    error.value = err.message || 'An error occurred'
   288→    extraGroups.value = []
   289→  } finally {
   290→    loading.value = false
   291→  }
   292→}
   293→
   294→const fetchAvailableExtras = async () => {
   295→  try {
   296→    const response = await api.get('/library/extras')
   297→    if (response.success) {
   298→      // Ensure we always set an array
   299→      if (Array.isArray(response.data)) {
   300→        availableExtras.value = response.data
   301→      } else if (response.data && Array.isArray(response.data.extras)) {
   302→        availableExtras.value = response.data.extras
   303→      } else {
   304→        availableExtras.value = []
   305→      }
   306→    }
   307→  } catch (err) {
   308→    console.error('Failed to fetch extras', err)
   309→    availableExtras.value = []
   310→  }
   311→}
   312→
   313→const saveGroup = async () => {
   314→  saving.value = true
   315→  try {
   316→    if (editingGroup.value) {
   317→      const response = await api.put(`/library/extra-groups/${editingGroup.value.id}`, formData.value)
   318→      if (response.success) {
   319→        await fetchExtraGroups()
   320→        closeModal()
   321→      } else {
   322→        alert('Failed to update extra group')
   323→      }
   324→    } else {
   325→      const response = await api.post('/library/extra-groups', formData.value)
   326→      if (response.success) {
   327→        await fetchExtraGroups()
   328→        closeModal()
   329→      } else {
   330→        alert('Failed to create extra group')
   331→      }
   332→    }
   333→  } catch (err: any) {
   334→    alert(err.message || 'An error occurred')
   335→  } finally {
   336→    saving.value = false
   337→  }
   338→}
   339→
   340→const editGroup = (group: any) => {
   341→  editingGroup.value = group
   342→  formData.value = {

   345→    MinRequired: group.MinRequired,

   348→    DisplayOrder: group.DisplayOrder,
   349→    ExtraIds: group.Extras ? group.Extras.map((item: any) => item.ExtraId) : []
   350→  }
   351→}
   352→
   353→const deleteGroup = async (id: number) => {
   354→  if (!confirm('Are you sure you want to delete this extra group?')) return
   355→
   356→  try {
   357→    const response = await api.delete(`/library/extra-groups/${id}`)
   358→    if (response.success) {
   359→      await fetchExtraGroups()
   360→    } else {
   361→      alert('Failed to delete extra group')
   362→    }
   363→  } catch (err: any) {
   364→    alert(err.message || 'An error occurred')
   365→  }
   366→}
   367→
   368→const closeModal = () => {
   369→  showCreateModal.value = false
   370→  editingGroup.value = null
   371→  formData.value = {
   372→    Name: '',
   373→    Description: '',
   374→    MinRequired: 0,
   375→    MaxAllowed: 0,
   376→    AllowMultipleQuantities: true,
   377→    DisplayOrder: 0,
   378→    ExtraIds: []
   379→  }

   385→  }
