    26→    /// <summary>

    30→    public async Task<IActionResult> GetExtraGroups()
    31→    {
    32→        try

    34→            var userId = GetUserId();

    36→                .Where(g => g.UserId == userId && g.CatalogId == null && g.IsActive)
    37→                .Include(g => g.GroupItems)

    39→                .OrderBy(g => g.DisplayOrder)
    40→                .ThenBy(g => g.Name)

    43→                    g.Id,
    44→                    g.Name,
    45→                    g.Description,
    46→                    g.MinRequired,
    47→                    g.MaxAllowed,
    48→                    g.AllowMultipleQuantities,
    49→                    g.DisplayOrder,
    50→                    g.IsActive,
    51→                    g.CreatedAt,
    52→                    g.UpdatedAt,
    53→                    Extras = g.GroupItems.Select(gi => new
    54→                    {
    55→                        gi.Id,
    56→                        gi.ExtraId,
    57→                        gi.DisplayOrder,
    58→                        gi.PriceOverride,
    59→                        Extra = new
    60→                        {
    61→                            gi.Extra.Id,
    62→                            gi.Extra.Name,
    63→                            gi.Extra.Description,
    64→                            gi.Extra.Code,
    65→                            gi.Extra.BasePrice,
    66→                            gi.Extra.ImageUrl
    67→                        }
    68→                    }).OrderBy(e => e.DisplayOrder).ToList()
    69→                })
    70→                .ToListAsync();

    73→        }
    74→        catch (Exception ex)
    75→        {

   137→    /// <summary>
   138→    /// Create a new extra group
   139→    /// </summary>
   140→    [HttpPost]
   141→    public async Task<IActionResult> CreateExtraGroup([FromBody] CreateExtraGroupDto dto)
   142→    {
   143→        try
   144→        {
   145→            var userId = GetUserId();
   146→            var userIdString = userId.ToString();
   147→
   148→            var group = new CatalogItemExtraGroup
   149→            {
   150→                UserId = userId,
   151→                Name = dto.Name,
   152→                Description = dto.Description,
   153→                MinRequired = dto.MinRequired,
   154→                MaxAllowed = dto.MaxAllowed,
   155→                AllowMultipleQuantities = dto.AllowMultipleQuantities,
   156→                DisplayOrder = dto.DisplayOrder,
   157→                IsActive = true,
   158→                CreatedAt = DateTime.UtcNow,
   159→                UpdatedAt = DateTime.UtcNow,
   160→                CreatedBy = userIdString,
   161→                UpdatedBy = userIdString
   162→            };
   163→
   164→            _context.CatalogItemExtraGroups.Add(group);
   165→            await _context.SaveChangesAsync();
   166→
   167→            // Add extras to the group
   168→            if (dto.ExtraIds != null && dto.ExtraIds.Any())
   169→            {
   170→                var displayOrder = 0;
   171→                foreach (var extraId in dto.ExtraIds)
   172→                {
   173→                    var groupItem = new CatalogItemExtraGroupItem
   174→                    {
   175→                        ExtraGroupId = group.Id,
   176→                        ExtraId = extraId,
   177→                        DisplayOrder = displayOrder++,
   178→                        IsActive = true,
   179→                        CreatedAt = DateTime.UtcNow,
   180→                        UpdatedAt = DateTime.UtcNow,
   181→                        CreatedBy = userIdString,
   182→                        UpdatedBy = userIdString
   183→                    };
   184→                    _context.CatalogItemExtraGroupItems.Add(groupItem);
   185→                }
   186→                await _context.SaveChangesAsync();
   187→            }
   188→
   189→            // Reload with extras
   190→            var createdGroup = await _context.CatalogItemExtraGroups
   191→                .Where(g => g.Id == group.Id)
   192→                .Include(g => g.GroupItems)
   193→                    .ThenInclude(gi => gi.Extra)
   194→                .FirstOrDefaultAsync();
   195→
   196→            return CreatedAtAction(nameof(GetExtraGroup), new { id = group.Id },
   197→                new { success = true, data = createdGroup });
   198→        }
   199→        catch (Exception ex)
   200→        {
   201→            _logger.LogError(ex, "Error creating extra group");
   202→            return StatusCode(500, new { success = false, error = "Failed to create extra group" });
   203→        }
   204→    }
