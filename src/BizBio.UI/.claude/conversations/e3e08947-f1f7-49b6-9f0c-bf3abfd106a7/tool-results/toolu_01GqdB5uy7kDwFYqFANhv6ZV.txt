   199→<script setup lang="ts">
   200→definePageMeta({

   203→
   204→const api = useApi()
   205→const loading = ref(true)
   206→const error = ref('')
   207→const extras = ref<any[]>([])
   208→const showCreateModal = ref(false)
   209→const editingExtra = ref<any>(null)
   210→const saving = ref(false)
   211→
   212→const formData = ref({
   213→  Name: '',
   214→  Code: '',
   215→  BasePrice: 0,
   216→  Description: '',
   217→  ImageUrl: '',
   218→  DisplayOrder: 0
   219→})
   220→
   221→const fetchExtras = async () => {
   222→  loading.value = true
   223→  error.value = ''
   224→  try {
   225→    const response = await api.get('/library/extras')
   226→    if (response.success) {
   227→      // Ensure we always set an array
   228→      if (Array.isArray(response.data)) {
   229→        extras.value = response.data
   230→      } else if (response.data && Array.isArray(response.data.extras)) {
   231→        extras.value = response.data.extras
   232→      } else {
   233→        console.warn('Unexpected response structure:', response.data)
   234→        extras.value = []
   235→      }
   236→    } else {
   237→      error.value = 'Failed to load extras'
   238→      extras.value = []
   239→    }
   240→  } catch (err: any) {
   241→    error.value = err.message || 'An error occurred'
   242→    extras.value = []
   243→  } finally {
   244→    loading.value = false
   245→  }
   246→}
   247→
   248→const saveExtra = async () => {
   249→  saving.value = true
   250→  try {
   251→    if (editingExtra.value) {
   252→      const response = await api.put(`/library/extras/${editingExtra.value.Id}`, formData.value)
   253→      if (response.success) {
   254→        await fetchExtras()
   255→        closeModal()
   256→      } else {
   257→        alert('Failed to update extra')
   258→      }
   259→    } else {
   260→      const response = await api.post('/library/extras', formData.value)
   261→      if (response.success) {
   262→        await fetchExtras()
   263→        closeModal()
   264→      } else {
   265→        alert('Failed to create extra')
   266→      }
   267→    }
   268→  } catch (err: any) {
   269→    alert(err.message || 'An error occurred')
   270→  } finally {
   271→    saving.value = false
   272→  }
   273→}
   274→
   275→const editExtra = (extra: any) => {
   276→  editingExtra.value = extra
   277→  formData.value = {
   278→    name: extra.name,
   279→    code: extra.code || '',
   280→    basePrice: extra.basePrice,
   281→    description: extra.description || '',
   282→    imageUrl: extra.imageUrl || '',
   283→    displayOrder: extra.displayOrder
   284→  }
   285→}
   286→
   287→const deleteExtra = async (id: number) => {
   288→  if (!confirm('Are you sure you want to delete this extra?')) return
   289→
   290→  try {
   291→    const response = await api.delete(`/library/extras/${id}`)
   292→    if (response.success) {
   293→      await fetchExtras()
   294→    } else {
   295→      alert('Failed to delete extra')
   296→    }
   297→  } catch (err: any) {
   298→    alert(err.message || 'An error occurred')
   299→  }
   300→}
   301→
   302→const closeModal = () => {
   303→  showCreateModal.value = false
   304→  editingExtra.value = null
   305→  formData.value = {
   306→    name: '',
   307→    code: '',
   308→    basePrice: 0,
   309→    description: '',
   310→    imageUrl: '',
   311→    displayOrder: 0
   312→  }
   313→}

   316→  fetchExtras()
