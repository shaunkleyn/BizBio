     6→// ==================== TYPE DEFINITIONS ====================
     7→
     8→interface Menu {
     9→  id: number
    10→  name: string
    11→  sections: MenuSection[]
    12→  options: OptionGroup[]  // ← Global options array
    13→  extras: ExtraGroup[]    // ← Global extras array
    14→}
    15→
    16→interface MenuSection {
    17→  id: number
    18→  name: string
    19→  items: MenuItem[]
    20→}
    21→
    22→interface MenuItem {
    23→  id: number
    24→  name: string
    25→  description?: string
    26→  variants: Variant[]
    27→  media?: Media
    28→}
    29→
    30→interface Variant {
    31→  id: number
    32→  name: string
    33→  price: number
    34→  oldPrice?: number
    35→  default: boolean
    36→  optionIndices: number[]  // ← References menu.options[index]
    37→  extraIndices: number[]   // ← References menu.extras[index]
    38→}
    39→
    40→interface OptionGroup {
    41→  id: number
    42→  name: string
    43→  label: string
    44→  minimumSelect: number  // Usually >= 1 (required)
    45→  maximumSelect: number
    46→  items: ModifierItem[]
    47→}
    48→
    49→interface ExtraGroup {
    50→  id: number
    51→  name: string
    52→  label: string
    53→  minimumSelect: number  // Usually 0 (optional)
    54→  maximumSelect: number
    55→  items: ModifierItem[]
    56→}
    57→
    58→interface ModifierItem {
    59→  id: number
    60→  name: string
    61→  price: number
    62→  default: boolean
    63→}
    64→
    65→interface Media {
    66→  baseUrl: string
    67→  rawImage: string
    68→}
    69→
    70→// ==================== CART TYPES ====================
    71→
    72→interface CartItem {
    73→  menuItemId: number
    74→  menuItemName: string
    75→  variantId: number
    76→  variantName: string
    77→  quantity: number
    78→  basePrice: number
    79→
    80→  // Selected modifiers
    81→  selectedOptions: SelectedOption[]
    82→  selectedExtras: SelectedExtra[]
    83→
    84→  // Calculated
    85→  itemTotal: number
    86→}
    87→
    88→interface SelectedOption {
    89→  groupId: number
    90→  groupName: string
    91→  optionId: number
    92→  optionName: string
    93→  price: number
    94→}
    95→
    96→interface SelectedExtra {
    97→  groupId: number
    98→  groupName: string
    99→  extraId: number
   100→  extraName: string
   101→  quantity: number
   102→  unitPrice: number
   103→  totalPrice: number
   104→}

   106→// ==================== CART STORE (Vue/Nuxt) ====================
   107→
   108→import { defineStore } from 'pinia'
   109→
   110→export const useCartStore = defineStore('cart', {
   111→  state: () => ({
   112→    items: [] as CartItem[],
   113→    menu: null as Menu | null,
   114→  }),
   115→
   116→  getters: {
   117→    cartSubtotal(): number {
   118→      return this.items.reduce((sum, item) => sum + item.itemTotal, 0)
   119→    },
   120→
   121→    cartTotal(): number {
   122→      const deliveryFee = 20 // Could be dynamic
   123→      return this.cartSubtotal + deliveryFee
   124→    },
   125→
   126→    cartItemCount(): number {
   127→      return this.items.reduce((sum, item) => sum + item.quantity, 0)
   128→    },
   129→  },
   130→
   131→  actions: {
   132→    /**
   133→     * Load menu from API
   134→     */
   135→    async loadMenu(slug: string) {
   136→      const response = await fetch(`/api/v2/menu/by-slug/${slug}`)
   137→      this.menu = await response.json()
   138→    },
   139→
   140→    /**
   141→     * Add item to cart with modifier selection
   142→     * This shows the MRD pattern in action!
   143→     */
   144→    async addItemToCart(menuItem: MenuItem, variant: Variant) {
   145→      if (!this.menu) throw new Error('Menu not loaded')
   146→
   147→      const selectedOptions: SelectedOption[] = []
   148→      const selectedExtras: SelectedExtra[] = []
   149→
   150→      // ==================== HANDLE OPTIONS (Required) ====================
   151→
   152→      // variant.optionIndices = [0, 1] means this variant uses:
   153→      // - menu.options[0]
   154→      // - menu.options[1]
   155→
   156→      for (const optionIndex of variant.optionIndices) {
   157→        const optionGroup = this.menu.options[optionIndex] // ← Direct array access!
   158→
   159→        // Show picker UI to user (modal, dropdown, etc.)
   160→        const selectedOption = await this.showOptionPicker(optionGroup)
   161→
   162→        if (!selectedOption) {
   163→          throw new Error(`${optionGroup.label} is required`)
   164→        }
   165→
   166→        selectedOptions.push({
   167→          groupId: optionGroup.id,
   168→          groupName: optionGroup.name,
   169→          optionId: selectedOption.id,
   170→          optionName: selectedOption.name,
   171→          price: selectedOption.price,
   172→        })
   173→      }
   174→
   175→      // ==================== HANDLE EXTRAS (Optional) ====================
   176→
   177→      // variant.extraIndices = [0, 1, 2] means this variant can have:
   178→      // - menu.extras[0]
   179→      // - menu.extras[1]
   180→      // - menu.extras[2]
   181→
   182→      for (const extraIndex of variant.extraIndices) {
   183→        const extraGroup = this.menu.extras[extraIndex] // ← Direct array access!
   184→
   185→        // Show optional extras picker (user can skip)
   186→        const selectedExtraItems = await this.showExtrasPicker(extraGroup)
   187→
   188→        selectedExtraItems.forEach((extra) => {
   189→          selectedExtras.push({
   190→            groupId: extraGroup.id,
   191→            groupName: extraGroup.name,
   192→            extraId: extra.item.id,
   193→            extraName: extra.item.name,
   194→            quantity: extra.quantity,
   195→            unitPrice: extra.item.price,
   196→            totalPrice: extra.item.price * extra.quantity,
   197→          })
   198→        })
   199→      }
   200→
   201→      // ==================== CALCULATE TOTAL ====================
   202→
   203→      const itemTotal = this.calculateItemTotal(
   204→        variant.price,
   205→        selectedOptions,
   206→        selectedExtras,
   207→        1 // quantity
   208→      )
   209→
   210→      // ==================== ADD TO CART ====================
   211→
   212→      const cartItem: CartItem = {
   213→        menuItemId: menuItem.id,
   214→        menuItemName: menuItem.name,
   215→        variantId: variant.id,
   216→        variantName: variant.name,
   217→        quantity: 1,
   218→        basePrice: variant.price,
   219→        selectedOptions,
   220→        selectedExtras,
   221→        itemTotal,
   222→      }
   223→
   224→      this.items.push(cartItem)
   225→    },
   226→
   227→    /**
   228→     * Calculate item total price
   229→     */
   230→    calculateItemTotal(
   231→      basePrice: number,
   232→      options: SelectedOption[],
   233→      extras: SelectedExtra[],
   234→      quantity: number
   235→    ): number {
   236→      let total = basePrice
   237→
   238→      // Add option costs
   239→      options.forEach((option) => {
   240→        total += option.price
   241→      })
   242→
   243→      // Add extras costs
   244→      extras.forEach((extra) => {
   245→        total += extra.totalPrice
   246→      })
   247→
   248→      return total * quantity
   249→    },
   250→
   251→    /**
   252→     * Update item quantity
   253→     */
   254→    updateQuantity(index: number, quantity: number) {
   255→      if (quantity <= 0) {
   256→        this.items.splice(index, 1)
   257→        return
   258→      }
   259→
   260→      const item = this.items[index]
   261→      item.quantity = quantity
   262→
   263→      // Recalculate total
   264→      item.itemTotal = this.calculateItemTotal(
   265→        item.basePrice,
   266→        item.selectedOptions,
   267→        item.selectedExtras,
   268→        quantity
   269→      )
   270→    },
   271→
   272→    /**
   273→     * Remove item from cart
   274→     */
   275→    removeItem(index: number) {
   276→      this.items.splice(index, 1)
   277→    },
   278→
   279→    /**
   280→     * Clear entire cart
   281→     */
   282→    clearCart() {
   283→      this.items = []
   284→    },
   285→
   286→    /**
   287→     * Validate cart with backend and get final totals
   288→     */
   289→    async validateCart() {
   290→      const requestBody = {
   291→        items: this.items.map((item) => ({
   292→          variantId: item.variantId,
   293→          quantity: item.quantity,
   294→          selectedOptions: item.selectedOptions.map((opt) => ({
   295→            optionId: opt.optionId,
   296→          })),
   297→          selectedExtras: item.selectedExtras.map((ext) => ({
   298→            extraId: ext.extraId,
   299→            quantity: ext.quantity,
   300→          })),
   301→        })),
   302→        deliveryFee: 20,
   303→      }
   304→
   305→      const response = await fetch('/api/v2/menu/calculate-cart', {
   306→        method: 'POST',
   307→        headers: { 'Content-Type': 'application/json' },
   308→        body: JSON.stringify(requestBody),
   309→      })
   310→
   311→      if (!response.ok) {
   312→        const error = await response.json()
   313→        throw new Error(error.error || 'Cart validation failed')
   314→      }
   315→
   316→      return await response.json()
   317→    },
