   318→function scrollToCategory(categoryId: number) {
   319→  const element = document.getElementById(`category-${categoryId}`)

   321→    // Account for both global header (64px) and sticky category tabs (~60px)

   323→    const tabsHeight = 60 // Category tabs
   324→    const offset = headerHeight + tabsHeight + 10 // Extra padding

   328→    window.scrollTo({
   329→      top: offsetPosition,
   330→      behavior: 'smooth'
