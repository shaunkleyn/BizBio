    28→    /// <summary>
    29→    /// Get all library items for the current user
    30→    /// </summary>
    31→    [HttpGet]
    32→    public async Task<IActionResult> GetLibraryItems([FromQuery] int? categoryId = null)
    33→    {
    34→        var userId = GetUserId();
    35→
    36→        var query = _context.CatalogItems
    37→            .Where(i => i.UserId == userId);
    38→
    39→        if (categoryId.HasValue)
    40→        {
    41→            query = query.Where(i => i.CategoryId == categoryId.Value);
    42→        }
    43→
    44→        var items = await query
    45→            .OrderBy(i => i.Id)
    46→            .Select(i => new
    47→            {
    48→                i.Id,
    49→                i.Name,
    50→                i.Description,
    51→                i.Price,
    52→                i.CategoryId,
    53→                i.ItemType,
    54→                i.BundleId,
    55→                images = ParseJsonArray(i.Images),
    56→                tags = ParseJsonArray(i.Tags),
    57→                i.SortOrder,
    58→                variants = i.Variants
    59→                    .Where(v => v.IsActive)
    60→                    .Select(v => new
    61→                    {
    62→                        v.Id,
    63→                        v.Title,
    64→                        v.Price,
    65→                        v.SizeValue,
    66→                        v.SizeUnit,
    67→                        v.IsDefault
    68→                    }).ToList(),
    69→                extraGroups = i.ExtraGroupLinks
    70→                    .Where(l => l.IsActive)
    71→                    .OrderBy(l => l.DisplayOrder)
    72→                    .Select(l => new
    73→                    {
    74→                        l.ExtraGroupId,
    75→                        l.ExtraGroup.Name,
    76→                        l.ExtraGroup.Description,
    77→                        l.ExtraGroup.MinRequired,
    78→                        l.ExtraGroup.MaxAllowed,
    79→                        l.ExtraGroup.AllowMultipleQuantities,
    80→                        Extras = l.ExtraGroup.GroupItems
    81→                            .Where(gi => gi.IsActive)
    82→                            .OrderBy(gi => gi.DisplayOrder)
    83→                            .Select(gi => new
    84→                            {
    85→                                gi.Extra.Id,
    86→                                gi.Extra.Name,
    87→                                gi.Extra.BasePrice,
    88→                                gi.PriceOverride
    89→                            }).ToList()
    90→                    }).ToList()
    91→            })
    92→            .ToListAsync();
    93→
    94→        return Ok(new { success = true, data = new { items } });
    95→    }

   405→    /// <summary>

   408→    [HttpPost("{id}/add-to-catalog")]
   409→    public async Task<IActionResult> AddToCatalog(int id, [FromBody] AddToCatalogDto dto)
   410→    {
   411→        var userId = GetUserId();
   412→
   413→        var libraryItem = await _context.CatalogItems
   414→            .Include(i => i.Variants)
   415→            .Include(i => i.ExtraGroupLinks.Where(l => l.IsActive))
   416→            .FirstOrDefaultAsync(i => i.Id == id && i.UserId == userId);
   417→
   418→        if (libraryItem == null)
   419→            return NotFound(new { success = false, error = "Library item not found" });
   420→
   421→        // Verify user owns the catalog
   422→        var catalog = await _context.Catalogs
   423→            .Include(c => c.Profile)
   424→            .FirstOrDefaultAsync(c => c.Id == dto.CatalogId);
   425→
   426→        if (catalog == null || catalog.Profile.UserId != userId)
   427→            return NotFound(new { success = false, error = "Catalog not found" });
   428→
   429→        // Create a copy of the item for the catalog
   430→        var catalogItem = new CatalogItem
   431→        {
   432→            CatalogId = dto.CatalogId,
   433→            CategoryId = dto.CategoryId ?? libraryItem.CategoryId,
   434→            ItemType = libraryItem.ItemType,
   435→            BundleId = libraryItem.BundleId,
   436→            Name = libraryItem.Name,
   437→            Description = libraryItem.Description,
   438→            Price = libraryItem.Price,
   439→            Images = libraryItem.Images,
   440→            Tags = libraryItem.Tags,
   441→            AvailableInEventMode = libraryItem.AvailableInEventMode,
   442→            EventModeOnly = libraryItem.EventModeOnly,
   443→            SortOrder = dto.SortOrder,
   444→            SourceLibraryItemId = libraryItem.Id,
   445→            IsActive = true,
   446→            CreatedAt = DateTime.UtcNow,
   447→            UpdatedAt = DateTime.UtcNow,
   448→            CreatedBy = userId.ToString(),
   449→            UpdatedBy = userId.ToString()
   450→        };
   451→
   452→        _context.CatalogItems.Add(catalogItem);
   453→        await _context.SaveChangesAsync();
   454→
   455→        // Copy variants
   456→        if (libraryItem.Variants.Any())
   457→        {
   458→            foreach (var sourceVariant in libraryItem.Variants.Where(v => v.IsActive))
   459→            {
   460→                var variant = new CatalogItemVariant
   461→                {
   462→                    CatalogItemId = catalogItem.Id,
   463→                    Title = sourceVariant.Title,
   464→                    Price = sourceVariant.Price,
   465→                    Cost = sourceVariant.Cost,
   466→                    SizeValue = sourceVariant.SizeValue,

   468→                    UnitOfMeasure = sourceVariant.UnitOfMeasure,
   469→                    Sku = sourceVariant.Sku,
   470→                    Barcode = sourceVariant.Barcode,

   472→                    WeightG = sourceVariant.WeightG,

   474→                    CreatedAt = DateTime.UtcNow,

   497→                    CreatedBy = userId.ToString(),
   498→                    UpdatedBy = userId.ToString()
   499→                };
   500→                _context.CatalogItemExtraGroupLinks.Add(link);
   501→            }
   502→            await _context.SaveChangesAsync();
   503→        }
   504→
   505→        return Ok(new { success = true, data = new { catalogItemId = catalogItem.Id } });
   506→    }
