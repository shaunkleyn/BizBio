   170→    [HttpPost("dev-verify-email")]
   171→    public async Task<IActionResult> DevVerifyEmail([FromBody] DevVerifyEmailDto dto)
   172→    {
   173→        // Only allow in Development environment
   174→        var environment = Environment.GetEnvironmentVariable("ASPNETCORE_ENVIRONMENT");
   175→        if (environment != "Development")
   176→        {
   177→            return NotFound();
   178→        }
   179→
   180→        try
   181→        {
   182→            if (string.IsNullOrEmpty(dto.Email))
   183→            {
   184→                return BadRequest(new { message = "Email is required" });
   185→            }
   186→
   187→            _logger.LogWarning("DEV ONLY: Manual email verification for {Email}", dto.Email);
   188→
   189→            // Get user by email and manually verify
   190→            var user = await _authService.GetUserByEmailAsync(dto.Email);
   191→            if (user == null)
   192→            {
   193→                return NotFound(new { message = "User not found" });
   194→            }
   195→
   196→            if (user.EmailVerified)
   197→            {
   198→                return Ok(new { message = "Email already verified" });
   199→            }
   200→
   201→            await _authService.ManuallyVerifyEmailAsync(dto.Email);
   202→
   203→            _logger.LogInformation("DEV ONLY: Email verified successfully for {Email}", dto.Email);
   204→
   205→            return Ok(new { message = "Email verified successfully! (Development bypass)" });
   206→        }
   207→        catch (Exception ex)
   208→        {
   209→            _logger.LogError(ex, "DEV ONLY: Error during manual email verification for {Email}", dto.Email);
   210→            return StatusCode(500, new { message = ex.Message });
