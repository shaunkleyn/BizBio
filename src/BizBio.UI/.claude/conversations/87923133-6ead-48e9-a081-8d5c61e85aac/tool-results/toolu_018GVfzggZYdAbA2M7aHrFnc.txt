    40→    public async Task SendEmailAsync(string toEmail, string subject, string htmlBody)
    41→    {
    42→        var stopwatch = Stopwatch.StartNew();
    43→
    44→        try
    45→        {
    46→            _logger.LogInformation("Attempting to send email to {Email} with subject: {Subject}", toEmail, subject);
    47→            _logger.LogDebug("SMTP Host: {Host}:{Port}, User: {User}", _smtpHost, _smtpPort, _smtpUsername);
    48→
    49→            // Track dependency for SMTP call
    50→            var dependency = new DependencyTelemetry
    51→            {
    52→                Name = "SMTP Email Send",
    53→                Type = "SMTP",
    54→                Target = $"{_smtpHost}:{_smtpPort}",
    55→                Data = subject
    56→            };
    57→
    58→            using var smtpClient = new SmtpClient(_smtpHost, _smtpPort)
    59→            {
    60→                EnableSsl = true,
    61→                Credentials = new NetworkCredential(_smtpUsername, _smtpPassword),
    62→                Timeout = 30000, // 30 seconds
    63→                DeliveryMethod = SmtpDeliveryMethod.Network,
    64→                UseDefaultCredentials = false
    65→            };
    66→
    67→            using var mailMessage = new MailMessage
    68→            {
    69→                From = new MailAddress(_fromEmail, _fromName),
    70→                Subject = subject,
    71→                Body = htmlBody,
    72→                IsBodyHtml = true
    73→            };
    74→
    75→            mailMessage.To.Add(toEmail);
    76→
    77→            await smtpClient.SendMailAsync(mailMessage);
    78→
    79→            _logger.LogInformation("Email sent successfully to {Email}", toEmail);
    80→
    81→            dependency.Success = true;
    82→            dependency.Duration = stopwatch.Elapsed;
    83→            _telemetryClient.TrackDependency(dependency);
    84→
    85→            _telemetryClient.TrackEvent("EmailSent", new Dictionary<string, string>
    86→            {
    87→                { "To", toEmail },
    88→                { "Subject", subject },
    89→                { "SmtpHost", _smtpHost }
    90→            });
    91→        }
    92→        catch (SmtpException smtpEx)
    93→        {
    94→            _logger.LogError(smtpEx, "SMTP error sending email to {Email}. Status: {Status}", toEmail, smtpEx.StatusCode);
    95→
    96→            var dependency = new DependencyTelemetry
    97→            {
    98→                Name = "SMTP Email Send",
    99→                Type = "SMTP",
   100→                Target = $"{_smtpHost}:{_smtpPort}",
   101→                Data = subject,
   102→                Success = false,
   103→                Duration = stopwatch.Elapsed
   104→            };
   105→            _telemetryClient.TrackDependency(dependency);
   106→
   107→            _telemetryClient.TrackException(smtpEx, new Dictionary<string, string>
   108→            {
   109→                { "Operation", "SendEmail" },
   110→                { "To", toEmail },
   111→                { "Subject", subject },
   112→                { "SmtpStatusCode", smtpEx.StatusCode.ToString() }
   113→            });
   114→            throw;
   115→        }
   116→        catch (Exception ex)
   117→        {
   118→            _logger.LogError(ex, "Failed to send email to {Email}", toEmail);
   119→
   120→            _telemetryClient.TrackException(ex, new Dictionary<string, string>
   121→            {
   122→                { "Operation", "SendEmail" },
   123→                { "To", toEmail },
   124→                { "Subject", subject }
   125→            });
   126→            throw;
   127→        }
   128→    }
