     1→import axios, { type AxiosInstance } from 'axios'
     2→
     3→let apiClient: AxiosInstance | null = null
     4→
     5→export const useApi = () => {
     6→  const config = useRuntimeConfig()
     7→  const router = useRouter()
     8→
     9→  if (apiClient) {
    10→    return apiClient
    11→  }
    12→
    13→  apiClient = axios.create({
    14→    baseURL: config.public.apiUrl as string,
    15→    headers: {
    16→      'Content-Type': 'application/json'
    17→    },
    18→    withCredentials: false,
    19→  })
    20→
    21→  // Add request interceptor to include JWT token
    22→  apiClient.interceptors.request.use(
    23→    (config) => {
    24→      if (import.meta.client) {
    25→        const token = localStorage.getItem('token')
    26→        if (token) {
    27→          config.headers.Authorization = `Bearer ${token}`
    28→        }
    29→
    30→        // Add start time for tracking
    31→        config.metadata = { startTime: Date.now() }
    32→      }
    33→      return config
    34→    },
    35→    (error) => {
    36→      return Promise.reject(error)
    37→    }
    38→  )
    39→
    40→  // Add response interceptor for error handling
    41→  apiClient.interceptors.response.use(
    42→    (response) => {
    43→      // Track successful API calls
    44→      if (import.meta.client && response.config.metadata?.startTime) {
    45→        try {
    46→          const duration = Date.now() - response.config.metadata.startTime
    47→          const method = response.config.method?.toUpperCase() || 'GET'
    48→          const url = `${response.config.baseURL}${response.config.url}`
    49→
    50→          // Use window.$nuxt to avoid SSR issues
    51→          if (typeof window !== 'undefined' && (window as any).$nuxt?.$trackDependency) {
    52→            (window as any).$nuxt.$trackDependency(
    53→              `${method} ${response.config.url}`,
    54→              method,
    55→              url,
    56→              duration,
    57→              true,
    58→              response.status,
    59→              {
    60→                responseSize: JSON.stringify(response.data).length,
    61→                endpoint: response.config.url
    62→              }
    63→            )
    64→          }
    65→        } catch (err) {
    66→          // Fail silently if tracking fails
    67→          console.debug('Tracking error:', err)
    68→        }
    69→      }
    70→      // Return only the data property instead of the full response
    71→      return response.data
    72→    },
    73→    (error) => {
    74→      // Track failed API calls
    75→      if (import.meta.client && error.config?.metadata?.startTime) {
    76→        try {
    77→          const duration = Date.now() - error.config.metadata.startTime
    78→          const method = error.config.method?.toUpperCase() || 'GET'
    79→          const url = `${error.config.baseURL}${error.config.url}`
    80→          const statusCode = error.response?.status || 0
    81→
    82→          // Use window.$nuxt to avoid SSR issues
    83→          if (typeof window !== 'undefined' && (window as any).$nuxt?.$trackDependency) {
    84→            (window as any).$nuxt.$trackDependency(
    85→              `${method} ${error.config.url}`,
    86→              method,
    87→              url,
    88→              duration,
    89→              false,
    90→              statusCode,
    91→              {
    92→                errorMessage: error.message,
    93→                endpoint: error.config.url
    94→              }
    95→            )
    96→          }
    97→
    98→          // Track as exception for 5xx errors
    99→          if (statusCode >= 500 && typeof window !== 'undefined' && (window as any).$nuxt?.$trackException) {
   100→            (window as any).$nuxt.$trackException(new Error(`API Error: ${method} ${error.config.url} - ${statusCode}`), {
   101→              endpoint: error.config.url,
   102→              statusCode,
   103→              duration,
   104→              errorMessage: error.response?.data?.message || error.message
   105→            })
   106→          }
   107→        } catch (err) {
   108→          // Fail silently if tracking fails
   109→          console.debug('Tracking error:', err)
   110→        }
   111→      }
   112→
   113→      // Only redirect to login if:
   114→      // 1. We got a 401 error
   115→      // 2. It's NOT from the login endpoint
   116→      // 3. It's NOT from the register endpoint
   117→      const isAuthEndpoint = error.config?.url?.includes('/auth/login') ||
   118→                            error.config?.url?.includes('/auth/register')
   119→
   120→      if (error.response?.status === 401 && !isAuthEndpoint && import.meta.client) {
   121→        // Token expired or invalid, redirect to login
   122→        localStorage.removeItem('token')
   123→        localStorage.removeItem('user')
   124→        router.push('/login')
   125→      }
   126→      return Promise.reject(error)
   127→    }
   128→  )
   129→
   130→  return apiClient
   131→}
